import torch
import torch.nn as nn
import torch.optim as optim
from torchvision import datasets, transforms
import matplotlib.pyplot as plt


print("="*60)
print("开始 MNIST 手写数字识别项目")
print(f"PyTorch 版本: {torch.__version__}")
print(f"是否使用GPU: {'是' if torch.cuda.is_available() else '否'} ({torch.cuda.get_device_name(0) if torch.cuda.is_available() else 'CPU'})")
print("="*60)

# 1. 准备数据（MNIST数据集会自动下载到当前文件夹的 data/ 目录下）
transform = transforms.ToTensor()
train_data = datasets.MNIST(root='./data', train=True, download=True, transform=transform)
test_data = datasets.MNIST(root='./data', train=False, download=True, transform=transform)

print(f"\n数据加载完成！")
print(f"训练样本数: {len(train_data)}")
print(f"测试样本数: {len(test_data)}")

# 2. 定义一个超简单的神经网络模型
class SimpleNN(nn.Module):
    def __init__(self):
        super().__init__()
        self.fc = nn.Linear(28*28, 10)  # MNIST图片是28x28，输出10个类别（0-9）
    
    def forward(self, x):
        x = x.view(-1, 28*28)  # 将图片展平
        return self.fc(x)

model = SimpleNN()
print(f"\n模型已创建: {model}")

# 3. 设置损失函数和优化器
criterion = nn.CrossEntropyLoss()
optimizer = optim.SGD(model.parameters(), lr=0.01)

# 4. 快速训练一个批次（为了演示，只训练一个批次，而不是完整轮次）
print("\n>>> 正在快速训练（1个批次）...")
data, target = train_data[0]  # 取第一个样本
data = data.unsqueeze(0)      # 增加批次维度
output = model(data)          # 前向传播
loss = criterion(output, target.unsqueeze(0))  # 计算损失

optimizer.zero_grad()  # 清零梯度
loss.backward()        # 反向传播
optimizer.step()       # 更新参数

print(f"第一个批次的损失: {loss.item():.4f}")

# 5. 可视化第一个训练样本
print("\n>>> 正在显示第一个训练样本...")
img, label = train_data[0]
plt.imshow(img.squeeze(), cmap='gray')
plt.title(f'标签: {label}')
plt.axis('off')
plt.show()

print("\n✅ MNIST 项目演示成功完成！")
print("下一步：你可以尝试增加更多的训练批次和轮次，以提高模型精度。")
